Codeforce 1325B - CopyCopyCopyCopyCopy

[題目敘述]:

[解法]

                            #include <bits/stdc++.h>
                            using namespace std;
                            int main() {
                                long long t,n,x;
                                cin>>t;
                                while(t--){
                                    cin>>n;
                                    set <int> s;
                                    for(int i=0;i<n;i++){
                                        cin>>x;
                                        s.insert(x);
                                    }
                                    cout<<s.size()<<endl;
                                }
                                
                            }
